        _.each(data, addDataParam);
      }
 
      xhr.open(method, endpoint, true);
      xhr.send(formData);
 
      return deferred.promise;

        formData.append(key, val);
      }

        var

        console.log(data);
        if (request.status === 200 || request.status === 201) {
          deferred.resolve(JSON.parse(request.responseText));

      }
